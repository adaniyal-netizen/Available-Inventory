        function openModal(id) {

            const item = returnData.find(i => i.id === id);

            if (!item || item.status === 'returned') return;


            currentActiveId = id;


            // Populate Modal

            document.getElementById('modal-title').innerText = item.name;

            document.getElementById('modal-sku').innerText = `SKU/Size: ${item.sku}`;

            document.getElementById('modal-qty').innerText = item.qty;

            document.getElementById('modal-barcode-img').src = item.barcode;


            // Show Modal

            document.getElementById('modal').classList.remove('hidden');

        }


        function closeModal() {

            document.getElementById('modal').classList.add('hidden');

            currentActiveId = null;

        }


        function markCurrentAsReturned() {

            if (currentActiveId) {

                const itemIndex = returnData.findIndex(i => i.id === currentActiveId);

                if (itemIndex > -1) {

                    returnData[itemIndex].status = 'returned';

                    renderCards();

                    closeModal();

                }

            }

        }


        function updateHeaderCount() {

            const pendingCount = returnData.filter(i => i.status === 'pending').length;

            const badge = document.querySelector('nav span');

            if(pendingCount === 0) {

                badge.className = "text-xs font-medium bg-green-100 text-green-700 px-2.5 py-1 rounded-full border border-green-200";

                badge.innerText = "All Complete";

            } else {

                badge.innerText = `${pendingCount} Items Pending`;

            }

        }
